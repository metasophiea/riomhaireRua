    //multi byte
        and:byte:byte:destinationByte       ->          15
        nand:byte:byte:destinationByte      ->          16
        or:byte:byte:destinationByte        ->          17
        nor:byte:byte:destinationByte       ->          18
        xor:byte:byte:destinationByte       ->          19

        convert:byte:inputMode:outputMode   ->          21

        inc:byte                            ->          23
        dec:byte                            ->          24
        neg:byte                            ->          25
    //multi byte
        add:byte:byte:destinationByte       ->          26

    0 - program counter (high)
    1 - program counter (low)
